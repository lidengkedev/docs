# CORS

app.all('*', function(req, res, next) {
    res.header("Access-Control-Allow-Origin", "*")
    // res.header("Access-Control-Allow-Headers", "*")
    res.header(
        "Access-Control-Allow-Headers", 
        "Origin, X-Requested-With, Accept, Content-Type, XFILENAME, XFILECATEGORY, XFILESIZE, token"
    )
    res.header("Access-Control-Allow-Methods","PUT,POST,GET,DELETE,OPTIONS")
    res.header("X-Powered-By",' 3.2.1')
    res.header("Content-Type", "application/json;charset=utf-8")
    //  res.header("Content-Type", "application/x-www-form-urlencoded;charset=utf-8")
    next()
});

# 创建可以提交GIT的命令行脚本工具

deploy.bat 脚本文件

@ECHO off
CHCP 65001
TITLE 命令行脚本工具
REM 新建设置版本的脚本文件 shell_version.js
ECHO const path = require('path');> shell_version.js
ECHO const fs = require('fs');>> shell_version.js
ECHO const pkg = require('./package.json');>> shell_version.js
ECHO function setVersion(version = '0.0.1') {>> shell_version.js
ECHO    let [a, b, c] = version.split('.').map(v =^> Number(v));>> shell_version.js
ECHO    c++;>> shell_version.js
ECHO    if (c ^> 999) { c = 0; b++; };>> shell_version.js
ECHO    if (b ^> 999) { b = 0; a++; };>> shell_version.js
ECHO    return `${a}.${b}.${c}`;>> shell_version.js
ECHO }>> shell_version.js
REM 新建重写 package.json 文件
ECHO function reloadConfigFile() {>> shell_version.js
ECHO    pkg.version = setVersion(pkg.version);>> shell_version.js
ECHO    fs.writeFileSync(path.join(__dirname, 'package.json'), JSON.stringify(pkg, null, 2));>> shell_version.js
ECHO }>> shell_version.js
ECHO reloadConfigFile();>> shell_version.js
COLOR 9
REM 开始设置版本号
ECHO ============================ 开始设置版本号 =========================
node shell_version.js
IF %ERRORLEVEL% EQU 1 (
    REM 异常信息
    ECHO ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
    ECHO ++++++++++++++++++++++++ 设置 版本 脚本运行错误 ++++++++++++++++++++++++
    ECHO ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
    PAUSE
)
ECHO ============================ NODEJS 脚本 提交 GIT COMMIT -M VERSION =======
ECHO const { execSync } = require('child_process');> shell_version.js
ECHO const pkg = require('./package.json');>> shell_version.js
ECHO execSync('git commit -m v'+pkg.version);>> shell_version.js
ECHO ============================ GIT ADD --A =================================
git add --a
ECHO ============================ GIT COMMIT -M DEPLOY ======================
node shell_version.js
DEL shell_version.js
ECHO ============================ GIT PUSH ORIGIN DEPLOY =======================
git push origin master
IF %ERRORLEVEL% EQU 1 (
    REM 异常信息
    COLOR 2
    ECHO +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
    ECHO ++++++++++++++ GIT CHECKOUT MASTER 脚本运行错误 ++++++++++++++++++++++++
    ECHO +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
    PAUSE
)
EXIT

# 编写一个一键启动 node 服务的脚本工具

start.sh 脚本工具

  1. 运行Node服务
  2. 打开浏览器并跳转到指定链接地址
#!/bin/bash node
echo start "C:\Program Files\Google\Chrome\Application\chrome.exe" http://localhost:3000/api.html
npm run server && start "C:\Program Files (x86)\Microsoft\Edge\Application\msedge.exe" http://localhost:3000/api.html
