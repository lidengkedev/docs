方式二

const fs = require('fs')
const path = require('path')

router.get('/download', (req, res, next) => {
    const { filename } = req.query
    const file_path = path.resolve(__dirname, `../../static/files/excel/${filename}.xls`)
    const state = fs.statSync(file_path)
    if (state.isFile()) {
        res.set({
            'content-disposition': `attachment;filename=${encodeURIComponent(filename)}.xls`,
            'content-type': 'application/octet-stream',
            'content-length': state.size
        })
        res.sendfile(fs.createReadStream(file_path).pipe(res))
    } else {
        res.end('file no exist!')
    }
})

方式三

const fs = require('fs')
const path = require('path')

router.get('/download', (req, res, next) => {
    const { filename } = req.query
    const file_path = path.resolve(__dirname, `../../static/files/excel/${filename}.xls`)
    const stream = fs.createReadStream(file_path)
    stream.on('data', chunk => {
        res.write(chunk, 'binary')
    })
    stream.on('end', () => {
        res.end()
    })
})

# 范例

/** 
 * 采用 有道词典翻译API 进行 文本转语音 
 * https://dict.youdao.com/dictvoice?audio=123&le=zh
 */
const options = {
    method: 'GET',
    hostname: 'dict.youdao.com',
    path: '/dictvoice?audio=123&le=zh'
}
// 调用http模块的request方法请求百度接口
const req = http.request(options, function (res) {
  const chunks = []

  res.on('data', function (chunk) {
    // 获取到的音频文件数据暂存到chunks里面
    chunks.push(chunk)
  })

  res.on('end', function () {
    // 这里用到了Buffer模块，大概意思就是把获取到的语音文件流存入到body里面，body是一个Buffer
    const body = Buffer.concat(chunks)
    // 生成的mp3文件存储的路径，文件名叫做iloveu.mp3
    const filePath = path.normalize(path.join(__dirname, './iloveu.mp3'))
    fs.writeFileSync(filePath, body)
  })
})
req.end()
