# 环境变量配置

# 定义请求地址
@baseUrl = http://localhost:3000/api

# 发起 GET 请求
GET {{baseUrl}}/user/auth HTTP/1.1

# 配置 token
@authToken = {{ auth.response.data.X-AuthToken }}

# 发起 GET 请求
GET {{baseUrl}}/user/edit HTTP/1.1
# 请求头配置 token
Authorization: {{ authToken }}
