// /modules/admin/controller.js
'use strict'

const chalk = require('chalk')
const BaseCommon = require('../commons/base-common')
const model = require('./model')

class AdminController extends BaseCommon {
  constructor() {
    super()
    this.login = this.login.bind(this)
    this.register = this.register.bind(this)
  }

  async login(req, res, next) {
    const { username, password } = req.body
    try {
      if (!username) {
        throw new Error('用户名参数错误')
      } else if (!password) {
        throw new Error('密码参数错误')
      }
    } catch (error) {
      console.log(chalk.red(error.message), error)
      res.send({
        status: -1,
        type: 'GET_ERROR_PARAM',
        message: error.message
      })
      return
    }
    const newpassword = this.encryption(password.toString())
    try {
      const user = await model.findOne({ username })
      if (!user) {
        res.send({
          status: -1,
          message: '用户不存在'
        })
      } else if (user.password !== newpassword.toString()) {
          console.log(chalk.red('用户登录密码错误'))
          res.send({
            status: -1,
            message: '该用户已存在，密码输入错误'
          })
      } else {
        const authorization = await this.authorization({
          id: user.id,
          name: user.usernam
        })
        await model.updateAuthorization({ authorization, id: user.id })
        res.send({
          status: 200,
          data: {
            authorization
          },
          message: '登录成功'
        })
      }
    } catch (error) {
        
    }
  }

  async register() {}
}

module.exports = new AdminController()

modules/admin/model.js

// /modules/admin/model.js
'use strict';

const { query } = require('../../mysql/index')

class AdminModel {
  constructor() {
    this.user_name = null;
    this.password = null;
    this.id = null;
    this.create_time = null;
    this.admin = '管理员';
    // 1: 普通管理员
    // 2: 超级管理员
    this.status = null;
    this.avatar = 'default.jpg';
    this.city = null;
  }
  async findOne({ username }) {
    const results = await query(`SELECT * FROM database.user WHERE username='${ username }'`)
    return results[0]
  }

  async create({ user_name, password, create_time, admin, status, avatar, city }) {
    const results =  await query(`INSERT INTO 
      database.user (
        user_name,
        password,
        create_time,
        admin,
        status,
        avatar,
        city
      ) 
      VALUES (
        '${user_name}',
        '${password}',
        '${create_time}',
        '${admin}',
        '${status}',
        '${avatar}',
        '${city}'
      )`)
    return results
  }

  async authorization({ authorization }) {
    const results = await query(`INSERT INTO database.user (authorization) VALUES ('${authorization}')`)
    return results
  }

  async updateAuthorization({ authorization, id }) {
    const results = await query(`UPDATE database.user
      SET authorization='${authorization}'
      WHERE id='${ id }'`)
    return results
  }
}

module.exports = new AdminModel()

modules/commons/base-common.js

// /modules/commons/base-common.js
'use strict'

var bcrypt = require('bcrypt')
const crypto = require('crypto')
var JWT = require('jsonwebtoken')

const SIGN_KEY = 'signKey_1890237910870'

class BaseCommon {
  constructor() {
    this.authorization = this.authorization.bind(this)
    this.checkAuthorization = this.checkAuthorization.bind(this)
    this.encryption = this.encryption.bind(this)
    this.Md5 = this.Md5.bind(this)
  }
  /**
   * 生成 token
   * @param {*} param0 
   * @returns 
   */
  async authorization({ id, name }) {
    const rules = { id, name }
    return new Promise((resolve, reject) => {
      JWT.sign(rules, SIGN_KEY, {
        expiresIn: 3600
      }, (err, authorization) => {
        if (err) return reject(err)
        resolve(authorization)
      })
    })
  }
  /**
   * 检查 token
   * @param {*} authorization 
   * @returns 
   */
  async checkAuthorization(authorization) {
    return new Promise((resolve, reject) => {
      JWT.verify(authorization, SIGN_KEY,
        (err, decoded) => {
        if (err) return reject(err)
        resolve(decoded)
      })
    })
  }

  encryption(password) {
    const newpassword = this.Md5(this.Md5(password).substr(2,7) + this.Md5(password))
    return newpassword
  }

  Md5(password) {
    const md5 = crypto.createHash('md5')
    return md5.update(password).digest('base64')
  }
}

module.exports = BaseCommon

db/modules/index.js

// db/modules/index.js
const Sequelize = require('sequelize')
const fs = require('fs')
const path = require('path')
const basename = path.basename(__filename)
const env = process.env.NODE_ENV || 'development'
const config = require(__dirname + '/../config/config.json')[env]
const db = {}
let sequelize
if (config.use_env_variable) {
  sequelize = new Sequelize(process.env[config.use_env_variable], config)
} else {
  sequelize = new Sequelize(config.database, config.username, config.password, config)
}
fs.readdirSync(__dirname).filter(file => {
  return (file.indexOf('.') !== 0) && (file !== basename) && (file.slice(-3) === '.js')
}).forEach(file => {
  const model = sequelize['import'](path.join(__dirname, file))
  db[model.name] = model
})
Object.keys(db).forEach(modelName => {
  if (db[modelName].associate) {
    db[modelName].associate(db)
  }
})
db.sequelize = sequelize
db.Sequelize = Sequelize
module.exports = db

app.js

// app.js
const express = require('express')
const bodyParser = require('body-parser')
const app = express()

const models = require('../db/models')

app.use(express.json());
app.use(express.urlencoded());
app.use(bodyParser.urlencoded({ extended: true }))

/** 查询任务列表 */
app.get('/list/:status/:page', async (req, res, next) => {
    try {
      let { status, page } = req.params;
      let limit = 10;
      let offset = (page - 1) * limit;
      let where = {};
      if (status != -1) {
        where.status = status
      }
      // 1. 状态 1：表示代办 2：完成 3：删除
      // 2. 分页
      let lists = await models.Table.findAndCountAll({ where, offset, limit })
      res.json({ status: 200, data: { lists }, message: 'success' })
    } catch (error) {
      
    }
});

/** 创建一个 */
app.post('/create', async (req, res, next) => {
  try {
    let { name, deadline, content } = req.body
    let tables = await models.Table.create({ name, deadline, content })
    res.json({ status: 200, data: { tables }, message: 'success' })
  } catch (error) {
    next(error)
  }
});

/** 修改一个 */
app.post('/update', async (req, res, next) => {
  try {
    let { id, name, deadline, content } = req.body
    let row = await models.Table.findOne({ where: { id } })
    if (row) {
      row = await row.update({ name, deadline, content })
    }
    res.json({ status: 200, data: { row }, message: 'success' })
  } catch (error) {
    next(error)
  }
});

/** 修改一个， 删除 */
app.post('/update_status', async (req, res, next) => {
  try {
    let { id, status } = req.body
    let row = await models.Table.findOne({ where: { id } })
    if (row && status != row.status) {
      row = await row.update({ status })
    }
    res.json({ status: 200, data: { row }, message: 'success' })
  } catch (error) {
    next(error)
  }
})

/** 错误监听 */
app.use((err, req, res, next) => {
    if (err) {
        res.status(500).json({
            message: err.message
        })
    }
})

const PORT = 3001
const HOSTNAME = '192.168.0.110'
app.listen(PORT, HOSTNAME, function() {
    console.log(`\nlistening to http://${HOSTNAME}:${PORT}/`)
})
